import java.sql.*;
import java.util.*;
class DbOpDemo
{
	Connection con;
	Statement stm;
	ResultSet rs;
	Scanner sc;

	DbOpDemo()
	{
	try{
	Driver d=new sun.jdbc.odbc.JdbcOdbcDriver();
	DriverManager.registerDriver(d);

	con=DriverManager.getConnection("jdbc:odbc:MyAccessDSN","admin","admin123");
	stm=con.createStatement(ResultSet.TYPE_SCROLL_SENSITIVE,ResultSet.CONCUR_UPDATABLE);
	rs=stm.executeQuery("Select * from Info");
	sc=new Scanner(System.in);
	}catch(Exception e)
	{
	e.printStackTrace();
	}
	}

	void display()
	{
	try{
	rs.beforeFirst();
	while(rs.next())
	{
	System.out.println("\nRoll : "+rs.getInt(1));
	System.out.println("Name : "+rs.getString(2));
	System.out.println("Percentage : "+rs.getFloat(3));
	}
	}
	catch(Exception e)
	{
	e.printStackTrace();
	}
	}

	void search()
	{
	try{
	System.out.println("Enter Roll Number : ");
	int roll=sc.nextInt();
	rs.beforeFirst();
	
	while(rs.next())
	{
	if(rs.getInt(1)==roll)
	{
	System.out.println("\nRoll : "+roll);
	System.out.println("Name : "+rs.getString(2));
	System.out.println("Percentage : "+rs.getFloat(3));
	return;
	}
	}
	System.out.println("Record Not Found...");
	}
	catch(Exception e)
	{
	e.printStackTrace();
	}
	}

	void delete()
	{
	try{
	System.out.println("Enter Roll Number : ");
	int roll=sc.nextInt();
	rs.beforeFirst();
	
	while(rs.next())
	{
	if(rs.getInt(1)==roll)
	{
	rs.deleteRow();
	System.out.println("Record Deleted...");
	return;
	}
	}
	System.out.println("Record Not Found...");
	
	}
	catch(Exception e)
	{
	e.printStackTrace();
	}
	}

	void update()
	{
	try{
	System.out.println("Enter roll : ");
	int roll=sc.nextInt();

	System.out.println("Enter Name : ");
	String nm=sc.next();

	System.out.println("Enter Percentage : ");
	float per=sc.nextFloat();

	rs.beforeFirst();
	while(rs.next())
	{
	if(rs.getInt(1)==roll)
	{
	rs.updateString(2,nm);
	rs.updateFloat(3,per);
	rs.updateRow();
	System.out.println("Record updated...");
	return;
	}
	}
	}
	catch(Exception e)
	{
	e.printStackTrace();
	}
	}

	void insert()
	{
	try{
	
	rs.last();
	int roll=rs.getInt(1)+1;

	System.out.println("Enter Name : ");
	String nm=sc.next();

	System.out.println("Enter Percentage : ");
	float per=sc.nextFloat();

	rs.moveToInsertRow();
	rs.updateInt(1,roll);
	rs.updateString(2,nm);
	rs.updateFloat(3,per);
	rs.insertRow();
	System.out.println("Record Inserted...");
	}
	catch(Exception e)
	{
	e.printStackTrace();
	}
	}

	public static void main(String... args)
	{
	DbOpDemo db;
	try{
	db=new DbOpDemo();
	
	while(true)
	{
	System.out.println("\n1.Display\n2.Search\n3.Insert\n4.Delete\n5.Update\n6.Exit\nEnter Option : ");
	int op=db.sc.nextInt();
	switch(op)
	{
	case 1:db.display();break;
	case 2:db.search();break;
	case 3:db.insert();break;
	case 4:db.delete(); break;
	case 5:db.update();break;
	case 6:System.exit(0);
	}
	}
	}catch(Exception e)
	{
	e.printStackTrace();
	}
	}
}
